\begin{frame}{``Delegate'' vs. ``Surrender''}
    \begin{itemize}
        \item Delegate = send with a trust or commission to act for another. Temporary.
        \item Surrender = to yield, to cede, to give up, to grant. Permanent.
    \end{itemize}
\end{frame}
